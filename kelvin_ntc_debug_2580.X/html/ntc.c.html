40   double  Celsius2Fahrenheit(double  tc)
41   {
42        return  (  (9  *  tc  +  16000)  /  5);
43   }

46   //  Fahrenheit2Celsius
47   //
48  
49   double  Fahrenheit2Celsius(double  tf)
50   {
51        return  (  (5  *  tf  -  16000)  /  9);
52   }
