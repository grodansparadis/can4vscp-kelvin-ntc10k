22   void  writeEEPROM(  uint16_t  address,  uint8_t  data  ) 
23   {
24       
25   #if  defined(EEADRH)     
26           EEADRH  =  (  address &rsaquo &rsaquo   8  )  &  0xff;
27   #endif
28  
29           EEADR  =  address  &  0xff; //  Data  Memory  Address  to  write 
30           EEDATA  =  data;      //  Data  Memory  Value  to  write 
31           EECON1bits.EEPGD  =  0;   //  Point  to  data  memory 
32           EECON1bits.CFGS  =  0;    //  Access  program  FLASH  or  Data  EEPROM  Memory 
33           EECON1bits.WREN  =  1;    //  Enable  writes 
34  
35           //required  sequence 
36           INTCONbits.GIE  =  0;          //  Disable  interrupts 
37           EECON2  =  0x55;  //  Write  55h 
38           EECON2  =  0xaa;  //  Write  AAh 
39           EECON1bits.WR  =  1;            //  Set  WR  bit  to  begin  write
40           INTCONbits.GIE  =  1;          //  Enable  Interrupts
41  
42   #if  defined(__18F25K80)  ||  defined(__18F26K80)  ||  defined(__18F45K80)  ||  defined(__18F46K80)  ||  defined(__18F65K80)  ||  defined(__18F66K80)
43        while  (!PIR4bits.EEIF);  //  wait  for  interrupt  to  signal  write  complete
44        PIR4bits.EEIF  =  0;       //  clear  EEPROM  write  operation  interrupt  flag
45   #else
46        while  (!PIR2bits.EEIF);  //  wait  for  interrupt  to  signal  write  complete
47        PIR2bits.EEIF  =  0;       //  clear  EEPROM  write  operation  interrupt  flag
48   #endif
49           EECON1bits.WREN  =  0;    //  disable  writes  on  write  complete
50  
51   }
